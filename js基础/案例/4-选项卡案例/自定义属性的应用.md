//循环给每个li都绑定点击事件
for (var i = 0; i < navList.length; i++) {
    navList[i].myIndex = i;

    navList[i].onclick = function() {
        //this.myIndex就是当前点击的对象获取其之前绑定的自定义对象
        changeTab(this.myIndex);
    }
}

/*
遇到的问题：
只有js代码执行完成后，用户才能看到页面，
但在执行js代码时，循环执行完成后，i已经变成3了
用户点击li标签，执行点击事件绑定的方法，i此时为3，出现错误
*/
/*

navList[i].myIndex = i;

解决方法2：闭包解决方案
