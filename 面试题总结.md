4. 判断一个字符串中出现次数最多的字符；并统计此字符的个数

   ```js
   /*
    *countMaxChar:次数最多的字符
    */
   function countMaxChar(str){
      	var obj = {};
       str.split('').forEach((item)=>{
           if(!obj[item]){
               obj[item] = 1;
           }else{
               obj[item]++;
           }
       });
       var maxCount = 0,
           maxKey = '';
       for(var key in obj){
           if(maxCount < obj[key]){
               maxCount = obj[key];
               maxKey = key;
           }
       }
       return {
           'maxCount':maxCount,
           'maxKey':maxKey
       };
   }
   var str = 'xiaoming';
   console.log(countMaxChar(str));//{maxCount: 2, maxKey: "i"}
   ```

3. 计算出班级的平均数

   ```js
   //法1：使用reduce
   function average() {
       var a = arguments;
       var average = Array.prototype.reduce.call(a, (prev, next, currentIndex) => {
           if (currentIndex === a.length - 1) {
               return (prev + next) / a.length;
           }
           return prev + next
       });
       return average;
   ```
